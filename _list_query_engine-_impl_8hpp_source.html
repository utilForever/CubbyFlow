75template <typename T, size_t N>
76ClosestIntersectionQueryResult<T, N> ListQueryEngine<T, N>::ClosestIntersection(
77 const Ray<double, N>& ray,
78 const GetRayIntersectionFunc<T, N>& testFunc) const
79{
80 ClosestIntersectionQueryResult<T, N> best;
81
82 for (const auto& item : m_items)
83 {
84 if (double dist = testFunc(item, ray); dist < best.distance)
85 {
86 best.distance = dist;
87 best.item = &item;
88 }
89 }
90
